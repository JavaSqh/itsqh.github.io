Liunx常用命令

一、登录与注销

  1. sudo useradd lilei  //添加用户 (不能被立即使用，需设置密码 sudo passwd lilei)  

    1. sudo adduser lilei  //添加用户

      1. login  //登录或切换用户

        1. logout //注销用户（命令行）  exit(shell-退出控制台)

          1. shutdown -h 10  //10分钟后自动关机	shutdown -c  //取消

            1. halt(root用户)  //关闭所有进程后自动关机

              1. poweroff //同上

                1. shutdown -r 10 //十分钟后自动重启

                  1. init 6  //重启 （0-停机，1-单用户，2-多用户，3-完全多用户，4-图形化，5-安全模式，6-重启）

                    1. reboot  //重启

二、目录与文件

  1. pwd   //显示当前工作目录

    1. mkdir mydir  //创建工作目录

      1. cd mydir  //更改工作目录

        1. rmdir mydir //删除工作目录

          1. touch myfile  //创建文件

            1. mv myfile mydir  //移动目录或文件

              1. cp myfile myfir  //复制目录或文件

                1. rm -rf mydir  //删除目录或文件

                  1. ls -l myfile  //查看文件最后被编辑时间

                    1. ls -lu myfile //查看文件最后被访问时间

                      1. touch -at 01011212 myfile  //修改文件最后被访问时间

                        1. ls //列出所有文件和目录

                          1. ls -a //查看所有文件

                            1. ls -i //显示文件索引节点号

                              1. ls -l //详细显示

                                1. ls -m //以逗号分隔

                                  1. sudo apt-get install tree 

                                    1. tree -l//以树状图列出目录内容

                                      1. tree -a //所有

                                        1. tree -i //不以阶梯状

                                          1. tree -s  //列出文件或目录大小

                                            1. tree -t  //按更改时间

                                              1. file -b myfile  //显示目录或文件的详细信息

                                                1. stat myfile  //同上

三、文件内容显示

  1. cat > myfile  //创建文件并编辑内容（ctrl+D结束编辑）

    1. cat -n myfile  //查看文件

      1. chmod [u/g/o/a][+/-/=][r/w/x] myfile  //更改文件权限
        u-user,g-group,o-others,a-all . +-添加,–删除,=-重置 .
        r-read读（4），w-write写（2），x-execute执行（1）
        1. more myfile  //分页往后显示文件（Space空格）

          1. less myfile  //分页自由显示文件（Page Down / Page Up）

            1. head (-10) myfile  //指定显示文件前若干行（默认前10）

              1. tail (-10) myfile  //指定显示文件后若干行（默认后10）

四、文件内容处理

  1. sort myfile  //对文件内容进行排序

    1. sort -r myfile  //逆序

      1. uniq myfile  //检查文件中的重复内容

        1. grep （-c）‘a’ myfile  //在文件中查找指定内容 (显示行号)

          1. diff myfile01 myfile02  //对不同文件进行比较

            1. diff3 myfile01 myfile02 myfile03  //三个文件

              1. sdiff myfile01 myfile02  //合并

                1. cmp myfile01 myfile02  //通过字节对不同文件进行比较

                  1. comm myfile01 myfile02  //对有序文件进行比较

                    1. cut -b(-c)(-d) 2(3) myfile  //对文件内容进行剪切

                      1. paste myfile02 myfile01 //对文件内容进行粘贴 02-）01

                        1. wc （-参数） myfile  //对文件内容进行统计 （c-字符数,w-单词数,l-行数）

五、压缩

  1. zip myfile.zip myfile  //压缩

    1. zip -d myfile.zip myfile  //添加

      1. zip -m myfile.zip myfile  //删除

        1. unzip -o myfile.zip  //解压（覆盖）

          1. unzip -n myfile.zip  //解压（不覆盖）

            1. zipinfo myfile.zip  //列出压缩文件信息

  2. tar -xvf myfile.tar.gz//解压

六、获取帮助

  1. man ls  //获取帮助

    1. man -k ls  //不清楚完整名字

      1. whatis ls  //获取帮助

        1. help cd  / cd –help  //获取帮助 -d(简短描述) -s(用法简介)

          1. info who  //获取帮助

七、其他命令

  1. clear  //清楚屏幕信息

    1. echo xx  //显示文本  x=0  echo $x . echo -e \$x . echo $(pwd)

      1. date  //显示日期和时间（+%y 年  +%m 月  +%d日）

        1. cal  //显示当前日期  cal -y

          1. ps  //查看当前进程  -A(所有)  U  lilei (用户lilei)

            1. kill -9 2315  //终止某一进程  

              1. ps -ef | grep Jincheng

                1. pkill Jincheng

                  1. killall Jincheng

                    1. last  //显示最近登录系统的用户信息-6列

                      1. history （10） //显示历史指令-默认1000行

                        1. sudo adduser lilei sudo  //给普通用户赋予root权限

                          1. sudo usermod -G sudo lilei  //同上

                            1. alias l=’ls’  //定义命令别名

                              1. unalias l  //删除别名

                                1. alias  //列出别名
